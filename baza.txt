DROP DATABASE blog;

CREATE DATABASE blog;
USE blog; 

DROP TABLE posts;

CREATE TABLE posts (
                    id INT AUTO_INCREMENT,
                    title VARCHAR(555) NOT NULL,
                    body VARCHAR(2500) NOT NULL,
                    author VARCHAR(255) NOT NULL,
                    created_at TIMESTAMP,
                    PRIMARY KEY (id)
                    ); 
                    
DESCRIBE posts;
